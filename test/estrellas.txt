begin program estrellas;
{
	main() 
	var x,y,cont,red,green,blue:int;
	{	
		cont = 0;
		while (cont < 100)
		{
			x= random(-100,100);
			y = random(-100,100);
			red = random(0,255);
			green = random(0,255);
			blue = random(0,255);
			startFill(red,green,blue);
			star(x,y,4);
			arc(x,y,x,y);
			stopFill();
			cont = cont + 1;
		};
	};
	endfunc	
};
end
